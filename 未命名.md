```plantUML

@startuml
title Campus Second-hand Trading - Make Payment Activity Diagram

start

partition 学生用户 {
    :浏览商品;
    :选择商品;
    :提交订单;
    :发起支付;
}

partition 平台系统 {
    :生成订单;
    :校验订单信息;
    :请求第三方支付;
}

partition 第三方支付机构 {
    :处理支付请求;
    if (支付成功?) then (Yes)
        :返回支付成功结果;
    else (No)
        :返回支付失败结果;
    endif
}

partition 平台系统 {
    if (支付成功?) then (Yes)
        :更新订单状态为已支付;
        :通知卖家发货/交付;
    else (No)
        :提示支付失败;
        :允许重新支付;
    endif
}

partition 学生用户 {
    if (支付成功?) then (Yes)
        :确认收货;
        :提交评价;
    else (No)
        :重新发起支付或取消订单;
    endif
}

stop

@enduml
```
